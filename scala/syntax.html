val list = List(1, 2, 3, 4, 5)  
def timesTwo(i: Int) = i * 2

list.map(x => timesTwo(x))  // : List[Int] = List(2, 4, 6, 8, 10)        
list.map(timesTwo(_))       // : List[Int] = List(2, 4, 6, 8, 10)
list.map(timesTwo)          // : List[Int] = List(2, 4, 6, 8, 10)
list map timesTwo           // : List[Int] = List(2, 4, 6, 8, 10)
